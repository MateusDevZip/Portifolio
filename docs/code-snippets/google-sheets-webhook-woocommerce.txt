function importarPedidosWooCommerce() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Pedidos");
  const url = "https://certificadopj.com.br/wp-json/wc/v3/orders?per_page=100";
  const consumerKey = "ck_cf4c739e9b336e6e90bd294c3cf33cc5fb947af4";
  const consumerSecret = "cs_5e7a1fddd84a70689e09f04aa7bf60ad124e58eb";

  const options = {
    method: "get",
    headers: {
      Authorization: "Basic " + Utilities.base64Encode(consumerKey + ":" + consumerSecret)
    }
  };

  const response = UrlFetchApp.fetch(url, options);
  const pedidos = JSON.parse(response.getContentText());

  const lastRow = sheet.getLastRow();
  const existingOrders = {};
  const rowsToDelete = [];

  // Carrega os pedidos existentes na planilha
  if (lastRow > 1) {
    const orderNumbers = sheet.getRange(2, 2, lastRow - 1, 1).getValues();
    const voucherStatus = sheet.getRange(2, 1, lastRow - 1, 1).getValues();

    orderNumbers.forEach((row, index) => {
      const pedidoId = row[0];
      const isVoucherChecked = voucherStatus[index][0];
      if (pedidoId) {
        existingOrders[pedidoId] = { row: index + 2, isVoucherChecked: isVoucherChecked };
      }
    });
  }

  pedidos.forEach(pedido => {
    // Só processar pedidos com status confirmado
    if (pedido.status !== "processing" && pedido.status !== "completed") return;

    const pedidoId = pedido.id;
    if (existingOrders[pedidoId] && !existingOrders[pedidoId].isVoucherChecked) {
      rowsToDelete.push(existingOrders[pedidoId].row);
    }
  });

  // Remove linhas antigas de pedidos sem voucher
  rowsToDelete.sort((a, b) => b - a).forEach(row => sheet.deleteRow(row));

  pedidos.forEach(pedido => {
    if (pedido.status !== "processing" && pedido.status !== "completed") return;

    if (existingOrders[pedido.id] && existingOrders[pedido.id].isVoucherChecked) {
      return; // Não sobrescreve pedidos com voucher marcado
    }

    sheet.insertRowBefore(2);

    sheet.getRange(2, 1).setValue(false);
    sheet.getRange(2, 2).setValue(pedido.id);
    sheet.getRange(2, 3).setValue(new Date(pedido.date_created));
    sheet.getRange(2, 4).setValue(pedido.status);
    sheet.getRange(2, 5).setValue(pedido.billing.first_name);
    sheet.getRange(2, 6).setValue(pedido.billing.last_name);
    sheet.getRange(2, 7).setValue(pedido.billing.email);
    sheet.getRange(2, 8).setValue(pedido.meta_data.find(m => m.key === "_billing_cnpj")?.value || "");
    sheet.getRange(2, 9).setValue(pedido.billing.company);
    sheet.getRange(2, 10).setValue(pedido.meta_data.find(m => m.key === "_billing_cellphone")?.value || pedido.billing.phone);
    sheet.getRange(2, 11).setValue(pedido.billing.phone);
    sheet.getRange(2, 12).setValue(pedido.billing.address_1);
    sheet.getRange(2, 13).setValue(pedido.billing.city);
    sheet.getRange(2, 14).setValue(pedido.billing.state);
    sheet.getRange(2, 15).setValue(pedido.billing.postcode);

    // Campos personalizados
    sheet.getRange(2, 16).setValue(pedido.meta_data.find(m => m.key === "_billing_contador")?.value || "");
    sheet.getRange(2, 17).setValue(pedido.meta_data.find(m => m.key === "_billing_notas_emitidas")?.value || "");
    sheet.getRange(2, 18).setValue(pedido.meta_data.find(m => m.key === "_billing_trabalha_com")?.value || "");
    sheet.getRange(2, 19).setValue(pedido.meta_data.find(m => m.key === "billing_cnh")?.value || "");
  });

  // Reorganiza por data (coluna 3)
  if (sheet.getLastRow() > 1) {
    sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn())
         .sort({ column: 3, ascending: false });
  }
}
